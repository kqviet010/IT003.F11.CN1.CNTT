2. Selection Sort (Chỉ 1 vị trí sau đó duyện tìm min từ đó về sau, sau đó đổi chỗ)

void selectionSort(int a[], int n){
	int i_min;
	for (int i = 0; i < n-1; i++){
		i_min = i;
		for (int j = i+1; j < n; j++){
			if (a[j] < a[i_min])
				i_min = j;
		}
		swap(a[i], a[i_min]);
	}
}
